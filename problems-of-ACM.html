The Ants

题目链接

考虑蚂蚁相遇时，视为互相穿过，所以只需要计算每只蚂蚁的两向时间，进行比较得出答案即可。

#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

const int maxn = 1e6 + 10;
int t, l, n, x;

int main(){
    scanf("%d", &t);
    while (t--){
        int ans1 = 0, ans2 = 0;
        scanf("%d %d", &l, &n);
        for (int i = 1; i <= n; i++){
            scanf("%d", &x);
            ans1 = max(ans1, max(x, l - x));
            ans2 = max(ans2, min(x, l - x));
        }
        printf("%d %d\n", ans2, ans1);
    }
    return 0;
}

Lake Counting

题目链接

连通块个数判断。用dfs清楚连通块，统计dfs的次数即为答案。

#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

const int maxn = 1e2 + 10;
const int dx[10] = {0, -1, -1, -1, 0, 1, 1, 1, 0};
const int dy[10] = {0, -1, 0, 1, 1, 1, 0, -1, -1};
char ch;
int n, m, ans;
bool arr[maxn][maxn];

bool inMap(int x, int y){
    return 0 < x && x <= n && 0 < y && y <= m;
}

void dfs(int x, int y){
    arr[x][y] = 0;
    for (int i = 1; i <= 8; i++){
        int nx = x + dx[i];
        int ny = y + dy[i];
        if (inMap(nx, ny) && arr[nx][ny])
            dfs(nx, ny);
    }
}

int main(){
    scanf("%d %d", &n, &m);
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++){
            cin >> ch;
            if (ch == 'W')
                arr[i][j] = 1;
        }
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            if (arr[i][j]){
                ans++;
                dfs(i, j);
            }
    printf("%d\n", ans);
    return 0;
}
